\def\R{0.4}

\begin{tikzpicture}

    \draw[light=\T] (0,0) -- (\thetatwo:\raylen);
    \draw[angle=\T] (\arclen,0) arc (0:\thetatwo:\arclen) node[midway,right] {$\theta_T$};

    \draw[light=\R] (0,0) -- (-180-\thetaone:\raylen);
    \draw[angle=\R] (-\arclen,0) arc (-180:-180-\thetaone:\arclen) node[midway,left] {$\theta_R$};

    \draw (-0.1\raylen,0.8*\interfaceheight) node[left] {$n_1, v_1$};
